simple language based on c++

Типы:
	int, char, bool, void
	массив ?

Конструкции:
	цикл
	if
	функции
	блок

спец слова:
	read
	write
	return

------

пример:

example.pony
~~~~~~~~
int a;

int factorial1(int n) {
	if (n < 2) {
		return 1;
	} else {
		return n * factorial(n - 1);
	}
}

int factorial2(int n) {
	int i, ans;
	ans = 1;
	for (i = 2; i <= n; i = i + 1) {
		ans = ans * i;
	}
	return ans;
}

void main() {
	int a;
	read(a);
	write(factorial1(a));
	write(factorial2(a));
}
